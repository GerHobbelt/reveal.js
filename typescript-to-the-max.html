Important tsc Options

  • tsc -out [filename.js] | Output to Single File
  • tsc -c or --comment | Emit Comments
  • tsc -e or tsc --exec | Execute
  • tsc --target ES3 | Target your version of ECMAScript
  • tsc -h or tsc --help | Display Help
  • tsc --sourcemap | Generate corresponding .map file

Getting Extreme with TypeScript

Modules

							
module People {
    // visible
    exports interface IGreeter {
    	greeting:string;
    	greet: () => string;
    }
    
    // visible							
    exports class Greeter implements IGreeter {
        public greeting:string;
        private _name: string;
        
        // constructor
        constructor(msg:string) {
            this.greeting = message;
        }
        
        // get and set
        public get name():string {
        	return _name;
        }
        
        public set name(value:string) {
        	this._name = value;
        }
        
        // function
        greet() {
            return "Hello, " + this.greeting;
        }
    }
    
    // visible
    exports class Person extends Greeter {
    	constructor(msg:string) { super(msg); }
    	
    	introduce() {
    		return "My name is " + this.name();
    	}
    }
    
    // not visible
    class OtherPerson {
    	constructor() { }
    	
    	anotherIntroduction() {
    		return "My name is OtherPerson";
    	}
    }
}
